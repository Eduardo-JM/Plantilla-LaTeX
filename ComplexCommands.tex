% abstract
\makeatletter
	\renewenvironment{abstract}{%
		\if@twocolumn
			\chapter*{\abstractname}%
		\else %% <- here I've removed \small
			\begin{center}%
				{\normalfont\fontsize{16}{15}\bfseries\abstractname\vspace{\z@}}
			\end{center}%
		\quotation
		\fi
	}
	{
		\if@twocolumn\else\endquotation\fi
	}
